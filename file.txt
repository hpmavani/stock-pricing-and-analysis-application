//maximum subarray 

dp = {}
def fibonnaci(n):
    if n in dp:
        return dp[n]
    if n <= 1:
        return 1
    
    dp[n] = fibonnaci(n - 1) + fibonacci(n - 2)
    return dp[n]

def fib(n):
    if n <= 1:
        return 1
    a,b = 0, 1
    for _ in range(2, n + 1):
        a, b = b, a + b

    return b
